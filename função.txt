using System;

class Program
{
    static void Main()
    {
        int LerMoedas()
        {
            Console.Write("Digite o valor do equipamento: ");
            string entrada = Console.ReadLine();

            if (int.TryParse(entrada, out int valor))
            {
                return valor;
            }

            return 0;
        }

        int SomarEquipamentos(int valor1, int valor2)
        {
            return valor1 + valor2;
        }

        int arma1 = LerMoedas();
        int arma2 = LerMoedas();

        int resultado = SomarEquipamentos(arma1, arma2);

        Console.WriteLine($"\n🔨 Thorin diz: O custo total do conserto é {resultado} moedas de ouro!");
    }
}
